package main

import (
    "bytes"
    "log"
    "os"
    "os/exec"
)

func main() {
    // Prepare the go test command
    cmd := exec.Command("go", "test", "-v", "validcopy_test.go")

    // Use a bytes.Buffer to capture stdout and stderr
    var out bytes.Buffer
    cmd.Stdout = &out
    cmd.Stderr = &out

    // Run the command
    err := cmd.Run()
    if err != nil {
        log.Printf("Command execution error: %v", err)
    }

    // Output the results to stdout
    log.Println(out.String())

    // Write the results to a file
    outputFilename := "test_output.txt"
    if err := os.WriteFile(outputFilename, out.Bytes(), 0666); err != nil {
        log.Fatalf("Failed to write to output file: %v", err)
    }
}
